# Преобразование исходных данных
Сначала используем parser_* для парсинга html в json

После этого используем преобразование полученных json в целевую схему

Целевая схема выглядит следующим образом:
```json
[
...
    {
        "Модель": string,
        "Страна производитель": string,
        "Гарантия": int, # в месяцах
        "Операционная система": string,
        "Экран": {
            "Диагональ экрана": int, # в дюймах
            "Разрешение экрана": string,
            "Тип матрицы": string
        },
        "Габариты": {
            "Высота": string,
            "Ширина": string,
            "Глубина": string,
            "Вес": string
        },
        "Интерфейсы": {
            "Количество HDMI": int,
            "Количество USB": int,
            "Наличие Wi-Fi": bool
        },
        "Дополнительно": {
            "Smart TV": bool
        }
    },
    ...
]
